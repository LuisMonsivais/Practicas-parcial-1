import java.util.*;
import java.lang.Math;

public class JavaApplication {
    private static Scanner input = new Scanner(System.in);

    public static void main(String[] args) {
        System.out.println("Introduce el numerador de la fracción 1");
        double nFraccion1;
        
        nFraccion1 = input.nextDouble();
        System.out.println("Introduce el denominador de la fracción 1");
        double dFraccion1;
        
        dFraccion1 = input.nextDouble();
        System.out.println("Introduce el numerador de la fracción 2");
        double nFraccion2;
        
        nFraccion2 = input.nextDouble();
        System.out.println("Introduce el denominador de la fracción 2");
        double dFraccion2;
        
        dFraccion2 = input.nextDouble();
        double suma;
        
        suma = nFraccion1 * dFraccion2 + nFraccion2 * dFraccion1 / (dFraccion1 * dFraccion2);
        System.out.println("La suma es ");
        System.out.println(suma);
        double resta;
        
        resta = nFraccion1 * dFraccion2 - nFraccion2 * dFraccion1 / (dFraccion1 * dFraccion2);
        System.out.println("La resta es ");
        System.out.println(resta);
        double multiplicacion;
        
        multiplicacion = nFraccion1 * nFraccion2 / (dFraccion1 * dFraccion2);
        System.out.println("La multiplicación es ");
        System.out.println(multiplicacion);
        double division;
        
        division = dFraccion1 * nFraccion2 / (nFraccion1 * dFraccion2);
        System.out.println("La división es ");
        System.out.println(division);
    }
}
